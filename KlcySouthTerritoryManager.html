function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var action = e.parameter.action || 'getAll'; var result = {}; if (action === 'getAll') { result.territories = sheetToJSON(ss.getSheetByName('Territories')); result.assignments = sheetToJSON(ss.getSheetByName('Assignments')); result.comments = sheetToJSON(ss.getSheetByName('Comments')); } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); // Parse body — works whether Content-Type is application/json or text/plain var raw = e.postData ? e.postData.contents : '{}'; var data = JSON.parse(raw); var action = data.action; // ── ADD ASSIGNMENT ───────────────────────────────────────────── if (action === 'addAssignment') { var aSheet = ss.getSheetByName('Assignments'); var aId = aSheet.getLastRow(); // row-based auto ID aSheet.appendRow([ aId, data.territory_id, data.assigned_to, data.date_assigned, data.date_completed || '', data.service_year || '' ]); if (data.date_completed) { updateLastCompleted(ss, data.territory_id, data.date_completed); } } // ── COMPLETE ASSIGNMENT ──────────────────────────────────────── if (action === 'completeAssignment') { updateAssignmentRow(ss, data.territory_id, data.publisher, data.date_assigned, data.date_completed); updateLastCompleted(ss, data.territory_id, data.date_completed); } // ── UPDATE ASSIGNMENT (edit publisher / dates) ───────────────── if (action === 'updateAssignment') { var aS = ss.getSheetByName('Assignments'); var aD = aS.getDataRange().getValues(); for (var i = 1; i < aD.length; i++) { if (String(aD[i][1]) === String(data.territory_id) && String(aD[i][3]) === String(data.old_date_assigned) && String(aD[i][2]) === String(data.old_publisher)) { aS.getRange(i+1, 3).setValue(data.publisher); aS.getRange(i+1, 4).setValue(data.date_assigned); aS.getRange(i+1, 5).setValue(data.date_completed || ''); if (data.date_completed) { updateLastCompleted(ss, data.territory_id, data.date_completed); } break; } } } // ── DELETE ASSIGNMENT ────────────────────────────────────────── if (action === 'deleteAssignment') { var dS = ss.getSheetByName('Assignments'); var dD = dS.getDataRange().getValues(); for (var i = 1; i < dD.length; i++) { if (String(dD[i][1]) === String(data.territory_id) && String(dD[i][3]) === String(data.date_assigned) && String(dD[i][2]) === String(data.publisher)) { dS.deleteRow(i + 1); break; } } } // ── ADD / UPDATE TERRITORY ───────────────────────────────────── if (action === 'upsertTerritory') { var tS = ss.getSheetByName('Territories'); var tD = tS.getDataRange().getValues(); var found = false; for (var i = 1; i < tD.length; i++) { if (String(tD[i][0]) === String(data.territory_id)) { if (data.last_completed) tS.getRange(i+1, 2).setValue(data.last_completed); if (data.name !== undefined) tS.getRange(i+1, 3).setValue(data.name); found = true; break; } } if (!found) { tS.appendRow([data.territory_id, data.last_completed || '', data.name || '']); } } // ── DELETE TERRITORY (and its assignments + comments) ────────── if (action === 'deleteTerritory') { deleteRowsByTerritoryId(ss, 'Territories', data.territory_id, 0); deleteRowsByTerritoryId(ss, 'Assignments', data.territory_id, 1); deleteRowsByTerritoryId(ss, 'Comments', data.territory_id, 1); } // ── ADD COMMENT ──────────────────────────────────────────────── if (action === 'addComment') { var cS = ss.getSheetByName('Comments'); var cId = cS.getLastRow(); cS.appendRow([ cId, data.territory_id, data.assignment_id || '', data.comment_text, data.comment_date ]); } return ContentService .createTextOutput(JSON.stringify({status:'ok'})) .setMimeType(ContentService.MimeType.JSON); } // ── HELPERS ──────────────────────────────────────────────────────── function sheetToJSON(sheet) { if (!sheet) return []; var data = sheet.getDataRange().getValues(); var headers = data[0]; return data.slice(1).map(function(row) { var obj = {}; headers.forEach(function(h, i) { obj[h] = row[i]; }); return obj; }); } function updateLastCompleted(ss, tid, date) { if (!date) return; var sheet = ss.getSheetByName('Territories'); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { if (String(data[i][0]) === String(tid)) { sheet.getRange(i+1, 2).setValue(date); return; } } sheet.appendRow([tid, date, '']); } function updateAssignmentRow(ss, tid, publisher, dateAssigned, dateCompleted) { var sheet = ss.getSheetByName('Assignments'); var data = sheet.getDataRange().getValues(); for (var i = 1; i < data.length; i++) { // Match by territory_id, publisher, and date_assigned — all three for accuracy var rowTid = String(data[i][1]).trim(); var rowPub = String(data[i][2]).trim(); var rowDA = String(data[i][3]).trim(); var rowDC = data[i][4]; // Accept if date_completed is blank/null/empty (not yet completed) var notDone = !rowDC || rowDC === '' || rowDC === null; if (rowTid === String(tid) && rowPub === String(publisher) && rowDA === String(dateAssigned) && notDone) { sheet.getRange(i+1, 5).setValue(dateCompleted); return true; } } // Fallback: match only territory + publisher if date didn't match (date format edge case) for (var i = 1; i < data.length; i++) { var rowTid = String(data[i][1]).trim(); var rowPub = String(data[i][2]).trim(); var rowDC2 = data[i][4]; var notDone2 = !rowDC2 || rowDC2 === '' || rowDC2 === null; if (rowTid === String(tid) && rowPub === String(publisher) && notDone2) { sheet.getRange(i+1, 5).setValue(dateCompleted); return true; } } return false; } function deleteRowsByTerritoryId(ss, sheetName, tid, colIndex) { var sheet = ss.getSheetByName(sheetName); if (!sheet) return; var data = sheet.getDataRange().getValues(); for (var i = data.length - 1; i >= 1; i--) { if (String(data[i][colIndex]) === String(tid)) { sheet.deleteRow(i + 1); } } }
